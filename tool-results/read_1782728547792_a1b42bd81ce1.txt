    84→/* ─── Scroll reveal hook ─── */
    85→function useReveal() {
    86→  const ref = useRef<HTMLDivElement>(null);
    87→  useEffect(() => {
    88→    const el = ref.current;
    89→    if (!el) return;
    90→    const observer = new IntersectionObserver(
    91→      ([entry]) => { if (entry.isIntersecting) { el.classList.add('visible'); observer.unobserve(el); } },
    92→      { threshold: 0.1, rootMargin: '0px 0px -40px 0px' }
    93→    );
    94→    observer.observe(el);
    95→    return () => observer.disconnect();
    96→  }, []);
    97→  return ref;
    98→}

   830→      {showAdd && (
   831→        <div className="fixed inset-0 z-50 bg-black/50 flex items-center justify-center p-4" onClick={() => setShowAdd(false)}>
   832→          <Card className="w-full max-w-lg shadow-2xl" onClick={e => e.stopPropagation()}>
   833→            <CardHeader><CardTitle className="font-[family-name:var(--font-display)]">Ny faktura</CardTitle></CardHeader>
   834→            <CardContent className="space-y-4">
   835→              <div><label className="block text-sm font-medium mb-1">Kund</label><select value={form.customerId} onChange={e => setForm({ ...form, customerId: e.target.value })} className="w-full h-9 rounded-md border border-stone-300 px-3 text-sm"><option value="">Välj kund...</option>{customers.map(c => <option key={c.id} value={c.id}>{c.companyName}</option>)}</select></div>
   836→              <div><label className="block text-sm font-medium mb-1">Belopp (kr)</label><Input type="number" value={form.amount} onChange={e => setForm({ ...form, amount: e.target.value })} /></div>
   837→              <div><label className="block text-sm font-medium mb-1">Beskrivning</label><Input value={form.description} onChange={e => setForm({ ...form, description: e.target.value })} /></div>
   838→              <div><label className="block text-sm font-medium mb-1">Förfallodatum</label><Input type="date" value={form.dueDate} onChange={e => setForm({ ...form, dueDate: e.target.value })} /></div>
   839→              <div className="flex gap-3 pt-2"><Button onClick={saveInvoice} className="bg-stone-900 hover:bg-stone-800 text-white flex-1 active:scale-[0.97] transition-transform">Skapa faktura</Button><Button variant="outline" onClick={() => setShowAdd(false)}>Avbryt</Button></div>

   844→    </div>

   849→function MessagesSection({ messages, onRefresh }: { messages: ContactMsg[]; onRefresh: () => void }) {

   855→      {messages.length === 0 && <Card className="border-stone-200"><CardContent className="py-10 text-center text-stone-500">Inga meddelanden</CardContent></Card>}

   863→                <p className="text-stone-600 text-sm leading-relaxed">{m.message}</p>
   864→              </div>
   865→              <div className="flex gap-1 shrink-0">
   866→                <button onClick={() => toggleRead(m)} className="p-1.5 text-stone-400 hover:text-stone-700 rounded" title={m.read ? 'Markera oläst' : 'Markera läst'}>{m.read ? <Mail className="w-4 h-4" /> : <CheckCircle2 className="w-4 h-4" />}</button>

   868→              </div>
   869→            </div>
   870→          </CardContent>
   871→        </Card>
   872→      ))}
   873→    </div>
   874→  );
