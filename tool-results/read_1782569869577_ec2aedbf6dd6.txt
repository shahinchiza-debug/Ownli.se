   508→function LoginPage({ onLogin, onBack }: { onLogin: () => void; onBack: () => void }) {
   509→  const [email, setEmail] = useState('');
   510→  const [password, setPassword] = useState('');
   511→  const [showPw, setShowPw] = useState(false);
   512→  const [error, setError] = useState('');
   513→  const [loading, setLoading] = useState(false);
   514→
   515→  const handleLogin = async (e: React.FormEvent) => {
   516→    e.preventDefault();
   517→    setLoading(true);
   518→    setError('');
   519→    try {
   520→      const res = await fetch('/api/auth/login', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ email, password }) });
   521→      const data = await res.json();
   522→      if (res.ok) { onLogin(); }
   523→      else { setError(data.error || 'Inloggningsfel'); }
   524→    } catch { setError('Kunde inte ansluta till servern'); }
   525→    setLoading(false);
   526→  };
   527→
   528→  return (
   529→    <div className="min-h-screen flex items-center justify-center bg-gradient-to-br from-stone-900 via-stone-800 to-blue-900 px-4">
   530→      <div className="absolute inset-0 opacity-10" style={{ backgroundImage: 'radial-gradient(circle at 50% 50%, rgba(59,130,246,0.4) 0%, transparent 60%)' }} />
   531→      <Card className="w-full max-w-md relative z-10 shadow-2xl border-stone-200">
   532→        <CardHeader className="text-center space-y-4 pb-2">
   533→          <div className="flex items-center justify-center gap-2.5"><OwnliLogo size={40} markOnly /><span className="text-2xl font-bold text-stone-900">Ownli</span></div>
   534→          <p className="text-stone-500">Logga in på din admin-panel</p>
   535→        </CardHeader>
   536→        <CardContent>
   537→          <form onSubmit={handleLogin} className="space-y-4">
   538→            {error && <div className="bg-red-50 border border-red-200 text-red-700 text-sm rounded-lg p-3 flex items-center gap-2"><AlertCircle className="w-4 h-4 shrink-0" />{error}</div>}
   539→            <div><label className="block text-sm font-medium text-stone-700 mb-1.5">E-post</label><Input type="email" value={email} onChange={e => setEmail(e.target.value)} placeholder="admin@ownli.se" className="h-11" required /></div>
   540→            <div><label className="block text-sm font-medium text-stone-700 mb-1.5">Lösenord</label>
   541→              <div className="relative"><Input type={showPw ? 'text' : 'password'} value={password} onChange={e => setPassword(e.target.value)} placeholder="••••••••" className="h-11 pr-10" required /><button type="button" className="absolute right-3 top-1/2 -translate-y-1/2 text-stone-400 hover:text-stone-600" onClick={() => setShowPw(!showPw)}>{showPw ? <EyeOff className="w-4 h-4" /> : <Eye className="w-4 h-4" />}</button></div>
   542→            </div>

   544→          </form>
   545→          <div className="mt-4 text-center text-xs text-stone-400">Standard: admin@ownli.se / admin123</div>
